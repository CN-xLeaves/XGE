// 一个场景 
int CALLBACK MainScreen(int msg, int param, xge_event *eve)
{
	switch(msg)
	{
	case xge_msg_frame:			// 逻辑处理
		
		break;
	case xge_msg_draw:			//  绘图 
		
		break;
	case xge_msg_mouse_move:	//  鼠标移动 
		
		break;
	case xge_msg_mouse_down:	//  鼠标按下 
		
		break;
	case xge_msg_mouse_up:		//  鼠标弹起 
		
		break;
	case xge_msg_mouse_dclick:	//  鼠标双击 
		
		break;
	case xge_msg_mouse_whell:	//  鼠标滚轮滚动 
		
		break;
	case xge_msg_key_dowm:		//  键盘按键按下 
		
		break;
	case xge_msg_key_up:		//  键盘按键弹起 
		
		break;
	case xge_msg_key_repeat:	//  按键重复触发 
		
		break;
	case xge_msg_setfocus:		//  窗口获得焦点 
		
		break;
	case xge_msg_killfocus:		//  窗口失去焦点 
		
		break;
	case xge_msg_mouse_enter:	//  鼠标进入 
		
		break;
	case xge_msg_mouse_exit:	//  鼠标移出 
		
		break;
	case xge_msg_loadres:		//  加载资源 
		
		break;
	case xge_msg_freeres:		//  卸载资源 
		
		break;
	case xge_msg_close:			//  关闭窗口 
		return -1; 
		break;
	}
	return 0;
}
